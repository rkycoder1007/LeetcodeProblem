class Solution {
    public char findTheDifference(String s, String t) {
        char[] s1=s.toCharArray();
        char[] t1=t.toCharArray();
        Arrays.sort(s1);
        Arrays.sort(t1);
        int n=s1.length;
        for(int i=0;i<n;i++){
            if((s1[i]^t1[i])!=0)
                return t1[i];
        }
        return t1[n];
        
    }
}
